```swift
import Foundation

/// 미리보기 WKWebView에 인라인 주입할 로컬 웹 자산(현재 highlight.js).
/// Highlightr 패키지가 동봉하는 번들에서 1회 읽어 캐시한다. 못 찾으면 nil → 호출부가 CDN 폴백.
enum LocalWebAssets {
    static let highlightJS: String? = readBundleResource(named: "highlight.min.js")
    static let highlightCSSLight: String? = readBundleResource(named: "github.min.css")
    static let highlightCSSDark: String? = readBundleResource(named: "github-dark.min.css")

    /// 캐시된 자산으로 인라인 블록 생성(셋 다 있을 때만).
    static var hljsLocalBlock: String? {
        hljsBlock(js: highlightJS, cssLight: highlightCSSLight, cssDark: highlightCSSDark)
    }

    /// JS/CSS를 인라인 `<style>`·`<script>`로 감싼다(순수). 하나라도 nil이면 nil.
    /// 라이트/다크는 `<style media>`로 분기(원본 `<link media>`와 동일 동작).
    static func hljsBlock(js: String?, cssLight: String?, cssDark: String?) -> String? {
        guard let js, let cssLight, let cssDark else { return nil }
        return """
        <style media="(prefers-color-scheme: light)">\(cssLight)</style>
        <style media="(prefers-color-scheme: dark)">\(cssDark)</style>
        <script>\(js)</script>
        """
    }

    private static func readBundleResource(named name: String) -> String? {
        guard let url = locateInHighlightrBundle(named: name) else { return nil }
        return try? String(contentsOf: url, encoding: .utf8)
    }

    /// Highlightr 번들을 찾아(SyntaxHighlighter와 같은 루트들) 내부에서 파일명으로 탐색.
    /// 번들 내부 레이아웃 변동에 강건하도록 enumerate로 찾는다.
    private static func locateInHighlightrBundle(named name: String) -> URL? {
        let bundleName = "Highlightr_Highlightr.bundle"
        var roots: [URL] = []
        if let resources = Bundle.main.resourceURL { roots.append(resources) }
        roots.append(Bundle.main.bundleURL)
        if let exeDir = Bundle.main.executableURL?.deletingLastPathComponent() {
            roots.append(exeDir)
        }
        let fm = FileManager.default
        for root in roots {
            let bundle = root.appendingPathComponent(bundleName)
            guard fm.fileExists(atPath: bundle.path) else { continue }
            if bundle.appendingPathComponent(name).path.isEmpty == false,
               fm.fileExists(atPath: bundle.appendingPathComponent(name).path) {
                return bundle.appendingPathComponent(name)
            }
            if let en = fm.enumerator(at: bundle, includingPropertiesForKeys: nil) {
                for case let f as URL in en where f.lastPathComponent == name {
                    return f
                }
            }
        }
        return nil
    }
}
```

- [ ] **Step 5: hljsIncludes를 로컬 우선·CDN 폴백으로 수정**

`Sources/Services/MarkdownRenderer.swift`의 `hljsIncludes(theme:)`(~656)를 다음으로 교체. 기본 테마(github)면 로컬 인라인 블록을 쓰고, 로컬이 없거나 커스텀 테마면 기존 CDN. `hljs.highlightElement` 초기화 스크립트는 동일 유지:
```swift
    private func hljsIncludes(theme: String) -> String {
        let assets: String
        if theme == "github", let local = LocalWebAssets.hljsLocalBlock {
            // 기본 테마는 동봉된 Highlightr 번들을 인라인 주입(오프라인·네트워크 대기 해소).
            assets = local
        } else {
            // 로컬 자산이 없거나(극단) 커스텀 테마면 기존 CDN으로 폴백.
            let base = "https://cdn.jsdelivr.net/gh/highlightjs/cdn-release@11/build"
            let styleLinks: String
            switch theme {
            case "github":
                styleLinks = """
                <link rel="stylesheet" media="(prefers-color-scheme: light)" href="\(base)/styles/github.min.css">
                <link rel="stylesheet" media="(prefers-color-scheme: dark)" href="\(base)/styles/github-dark.min.css">
                """
            default:
                styleLinks = "<link rel=\"stylesheet\" href=\"\(base)/styles/\(htmlEscape(theme)).min.css\">"
            }
            assets = """
            \(styleLinks)
            <script src="\(base)/highlight.min.js"></script>
            """
        }

        return """
        \(assets)
        <script>
            document.addEventListener('DOMContentLoaded', function() {
                if (typeof hljs === 'undefined') return;
                document.querySelectorAll('pre code').forEach(function(el) {
                    if (el.classList.contains('language-mermaid')) return;
                    hljs.highlightElement(el);
                });
            });
        </script>
        """
    }
```

import XCTest
@testable import CmdMD

final class FileTreeBuildTests: XCTestCase {
    private func tempDir() -> URL {
        let d = URL(fileURLWithPath: NSTemporaryDirectory()).appendingPathComponent("ft-\(UUID().uuidString)")
        try? FileManager.default.createDirectory(at: d, withIntermediateDirectories: true)
        return d
    }

    func testBuildsTopLevelFoldersFirstThenFiles() throws {
        let root = tempDir(); defer { try? FileManager.default.removeItem(at: root) }
        try FileManager.default.createDirectory(at: root.appendingPathComponent("Bsub"), withIntermediateDirectories: true)
        try "x".write(to: root.appendingPathComponent("a.md"), atomically: true, encoding: .utf8)
        try "x".write(to: root.appendingPathComponent("z.md"), atomically: true, encoding: .utf8)

        let tree = AppState.buildFileTree(at: root, expanded: [])
        XCTAssertEqual(tree.map { $0.name }, ["Bsub", "a.md", "z.md"]) // 폴더 먼저, 이름순
    }

    func testExpandedFolderGetsChildrenUnexpandedDoesNot() throws {
        let root = tempDir(); defer { try? FileManager.default.removeItem(at: root) }
        let sub = root.appendingPathComponent("sub")
        try FileManager.default.createDirectory(at: sub, withIntermediateDirectories: true)
        try "x".write(to: sub.appendingPathComponent("inner.md"), atomically: true, encoding: .utf8)

        let collapsed = AppState.buildFileTree(at: root, expanded: [])
        XCTAssertEqual(collapsed.first?.children.isEmpty, true)

        let expanded = AppState.buildFileTree(at: root, expanded: [sub])
        XCTAssertEqual(expanded.first?.children.map { $0.name }, ["inner.md"])

- [ ] **Step 3: buildFileTree를 순수 static으로 전환**

`AppState.swift`의 `private func buildFileTree(at url: URL, depth: Int = 0) -> [FileTreeItem]`(~1110)를 아래로 교체한다. `expandedFolders` 인스턴스 참조를 `expanded` 인자로, 재귀에도 전달. catch는 인스턴스 `errorMessage`를 못 쓰므로 `[]` 반환(폴더 읽기 실패 시 빈 트리 — 드문 케이스, 토스트 생략):
```swift
    /// 폴더 트리를 만든다(순수·백그라운드 안전). `expanded`는 펼친 폴더 스냅샷.
    static func buildFileTree(at url: URL, expanded: Set<URL>, depth: Int = 0) -> [FileTreeItem] {
        guard depth < 10 else { return [] }
        do {
            let contents = try FileManager.default.contentsOfDirectory(
                at: url,
                includingPropertiesForKeys: [.isDirectoryKey],
                options: [.skipsHiddenFiles]
            )
            var items: [FileTreeItem] = []
            for itemURL in contents.sorted(by: { $0.lastPathComponent.localizedCaseInsensitiveCompare($1.lastPathComponent) == .orderedAscending }) {
                let resourceValues = try itemURL.resourceValues(forKeys: [.isDirectoryKey])
                let isDirectory = resourceValues.isDirectory ?? false
                if isDirectory {
                    let children = expanded.contains(itemURL) ? buildFileTree(at: itemURL, expanded: expanded, depth: depth + 1) : []
                    items.append(FileTreeItem(url: itemURL, isDirectory: true, isExpanded: expanded.contains(itemURL), children: children))
                } else if isListableInFileTree(itemURL) {
                    items.append(FileTreeItem(url: itemURL, isDirectory: false))
                }
            }
            return items.sorted { item1, item2 in
                if item1.isDirectory == item2.isDirectory {
                    return item1.name.localizedCaseInsensitiveCompare(item2.name) == .orderedAscending
                }
                return item1.isDirectory && !item2.isDirectory
            }
        } catch {
            return []
        }
    }
```
> `isListableInFileTree`는 이미 static(`Self.`/타입 호출). static 컨텍스트에서 `isListableInFileTree(itemURL)`로 호출 가능.

`loadFileTree()`(~1094)를 교체:
```swift
    func loadFileTree() {
        guard let folder = currentFolder else { return }
        // selectedFolder는 건드리지 않는다(펼치기·새로고침·이름변경 시 호출될 수 있음).
        let snapshot = expandedFolders
        fileTreeTask?.cancel()
        fileTreeTask = Task.detached(priority: .userInitiated) { [weak self] in
            let tree = AppState.buildFileTree(at: folder, expanded: snapshot)
            if Task.isCancelled { return }
            await MainActor.run { self?.fileTree = tree }
        }
    }
```
> Sendable: 컴파일러가 `FileTreeItem`/`[FileTreeItem]`의 Sendable을 문제 삼으면 `struct FileTreeItem`에 `: Sendable`을 추가한다(URL·Bool·UUID·[FileTreeItem]만 가진 값타입이라 안전). `Set<URL>`·`URL`은 이미 Sendable.
