> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 리더 위에 라이브러리 모드를 더한다 — 툴바 모드 토글(reader/library), 선택한 폴더의 직속 항목을 그리드/리스트로 보는 LibraryView, Finder식 폴더 선택(사이드바 행=선택/삼각형=펼침). 썸네일은 ②b로 분리.

**Tech Stack:** Swift 5.9 / SwiftUI(LazyVGrid·List·ContentUnavailableView, macOS 14+), XCTest.

import XCTest
@testable import CmdMD

final class LibraryListingTests: XCTestCase {
    private func tempDir() -> URL {
        let dir = URL(fileURLWithPath: NSTemporaryDirectory()).appendingPathComponent("lib-\(UUID().uuidString)")
        try? FileManager.default.createDirectory(at: dir, withIntermediateDirectories: true)
        return dir
    }

    func testEntriesReturnsFoldersAndFilesExcludingHidden() throws {
        let dir = tempDir(); defer { try? FileManager.default.removeItem(at: dir) }
        try "a".write(to: dir.appendingPathComponent("a.md"), atomically: true, encoding: .utf8)
        try "b".write(to: dir.appendingPathComponent("b.png"), atomically: true, encoding: .utf8)
        try FileManager.default.createDirectory(at: dir.appendingPathComponent("sub"), withIntermediateDirectories: true)
        try "h".write(to: dir.appendingPathComponent(".hidden"), atomically: true, encoding: .utf8)

        let entries = LibraryListing.entries(of: dir)
        XCTAssertEqual(Set(entries.map { $0.name }), ["a.md", "b.png", "sub"])
        XCTAssertEqual(entries.first { $0.name == "sub" }?.isDirectory, true)
        XCTAssertEqual(entries.first { $0.name == "a.md" }?.isDirectory, false)
    }

    func testEntriesEmptyForMissingFolder() {
        XCTAssertTrue(LibraryListing.entries(of: URL(fileURLWithPath: "/nope-\(UUID().uuidString)")).isEmpty)
    }

    func testEntriesSortedByParaLensFoldersFirstArchiveLast() throws {
        let dir = tempDir(); defer { try? FileManager.default.removeItem(at: dir) }
        try FileManager.default.createDirectory(at: dir.appendingPathComponent("10000_Projects"), withIntermediateDirectories: true)
        try FileManager.default.createDirectory(at: dir.appendingPathComponent("40000_Archive"), withIntermediateDirectories: true)
        try "z".write(to: dir.appendingPathComponent("z.md"), atomically: true, encoding: .utf8)
        try "a".write(to: dir.appendingPathComponent("a.md"), atomically: true, encoding: .utf8)

        let sorted = ParaLens.sorted(LibraryListing.entries(of: dir), under: dir).map { $0.name }
        // projects(폴더,rank0) → 기타 파일(rank3, 이름순) → archive(폴더,rank4)
        XCTAssertEqual(sorted, ["10000_Projects", "a.md", "z.md", "40000_Archive"])
    }
}

import Foundation

/// 라이브러리용 — 폴더의 직속 항목(폴더+파일)을 FileTreeItem으로 열거한다(읽기 전용).
/// 숨김 제외. 정렬은 호출부에서 ParaLens.sorted로 한다.
enum LibraryListing {
    static func entries(of folder: URL) -> [FileTreeItem] {
        let fm = FileManager.default
        guard let urls = try? fm.contentsOfDirectory(
            at: folder,
            includingPropertiesForKeys: [.isDirectoryKey],
            options: [.skipsHiddenFiles]
        ) else { return [] }
        return urls.map { url in
            let isDir = (try? url.resourceValues(forKeys: [.isDirectoryKey]))?.isDirectory ?? false
            return FileTreeItem(url: url, isDirectory: isDir)
        }
    }
}
```

import SwiftUI

/// 라이브러리 모드 — 선택한 폴더의 직속 항목을 그리드/리스트로 본다(읽기/탐색 전용).
struct LibraryView: View {
    @Environment(AppState.self) private var appState

    private var folder: URL? { appState.selectedFolder ?? appState.currentFolder }

    private var entries: [FileTreeItem] {
        guard let folder else { return [] }
        return ParaLens.sorted(LibraryListing.entries(of: folder), under: appState.currentFolder)
    }

    private var canGoUp: Bool {
        guard let f = folder, let root = appState.currentFolder else { return false }
        return f.standardizedFileURL.path != root.standardizedFileURL.path
    }

    var body: some View {
        VStack(spacing: 0) {
            header
            Divider()
            if folder == nil {
                ContentUnavailableView("폴더를 여세요", systemImage: "folder")
            } else if entries.isEmpty {
                ContentUnavailableView("이 폴더에 항목이 없습니다", systemImage: "tray")
            } else {
                content
            }
        }
    }

    private var header: some View {
        HStack(spacing: 8) {
            Button {
                if canGoUp, let f = folder {
                    appState.selectedFolder = f.deletingLastPathComponent()
                }
            } label: {
                Image(systemName: "chevron.up")
            }
            .buttonStyle(.borderless)
            .disabled(!canGoUp)

            Text(folder?.lastPathComponent ?? "Library")
                .font(.headline)
                .lineLimit(1)
            Spacer()
        }
        .padding(8)
    }

    @ViewBuilder
    private var content: some View {
        switch appState.libraryLayout {
        case .grid:
            ScrollView {
                LazyVGrid(columns: [GridItem(.adaptive(minimum: 110), spacing: 16)], spacing: 16) {
                    ForEach(entries) { item in cell(item) }
                }
                .padding(16)
            }
        case .list:
            List(entries) { item in
                cell(item)
            }
        }
    }

    private func cell(_ item: FileTreeItem) -> some View {
        let dim = ParaLens.classify(item.url, under: appState.currentFolder) == .archive
        return Group {
            if appState.libraryLayout == .grid {
                VStack(spacing: 6) {
                    Image(systemName: item.icon)
                        .font(.system(size: 34))
                        .frame(height: 44)
                    Text(item.name)
                        .font(.caption)
                        .lineLimit(2)
                        .multilineTextAlignment(.center)
                }
                .frame(maxWidth: .infinity)
                .padding(8)
            } else {
                HStack(spacing: 8) {
                    Image(systemName: item.icon)
                    Text(item.name).lineLimit(1)
                    Spacer()
                }
            }
        }
        .opacity(dim ? 0.45 : 1.0)
        .contentShape(Rectangle())
        .onTapGesture {
            if item.isDirectory {
                appState.selectFolderForLibrary(item.url)
            } else {
                appState.openDocument(at: item.url, inNewTab: true)
            }
        }
    }
}
```
