        XCTAssertEqual(ParaLens.classify(URL(fileURLWithPath: "/v/10000_Projects/Living_with_Damage/note.md")), .projects)
        XCTAssertEqual(ParaLens.classify(URL(fileURLWithPath: "/v/40000_Archive/2024/old.pdf")), .archive)

        XCTAssertEqual(ParaLens.classify(URL(fileURLWithPath: "/v/Downloads/x.md")), .other)
        XCTAssertEqual(ParaLens.classify(URL(fileURLWithPath: "/v/99_misc")), .other)
    }

    func testSortedOrdersParaRootsArchiveLast() {
        let items = [
            item("/v/40000_Archive"),
            item("/v/Inbox"),
            item("/v/10000_Projects"),
            item("/v/30000_Resources"),
            item("/v/20000_Areas"),
        ]
        let names = ParaLens.sorted(items).map { $0.name }
        XCTAssertEqual(names, ["10000_Projects", "20000_Areas", "30000_Resources", "Inbox", "40000_Archive"])
    }

    func testSortedFoldersBeforeFilesThenName() {
        let items = [
            item("/v/30000_Resources/b.md", dir: false),
            item("/v/30000_Resources/Zsub", dir: true),
            item("/v/30000_Resources/a.md", dir: false),
            item("/v/30000_Resources/Asub", dir: true),

        XCTAssertEqual(sorted.first?.id, parent.id)
        XCTAssertEqual(sorted.first?.children.first?.id, child.id)

```swift
import Foundation

/// 파일 트리 항목의 PARA 분류(숫자 접두사 규칙).
enum ParaCategory: Equatable {
    case projects, areas, resources, archive, other

    /// 정렬 순위 — Projects 먼저, Archive 맨 끝.
    var sortRank: Int {
        switch self {
        case .projects:  return 0
        case .areas:     return 1
        case .resources: return 2
        case .other:     return 3
        case .archive:   return 4
        }
    }
}

/// 사이드바 파일 트리를 PARA 렌즈로 정렬·분류하는 순수 헬퍼(표현 계층).
/// 데이터 모델·상태는 건드리지 않는다. RouteHelper/CleanupPlanner와 동형.
enum ParaLens {
    /// 경로 컴포넌트를 최상위→하위로 스캔해 처음 매칭되는 PARA 접두사로 분류.
    /// 캐스케이드: 최상위 PARA 루트가 하위 항목 전체의 분류를 결정한다.
    static func classify(_ url: URL) -> ParaCategory {
        for component in url.pathComponents {
            if let category = categoryForPrefix(component) {
                return category
            }
        }
        return .other
    }

    /// 한 폴더명 컴포넌트의 숫자 접두사를 분류로. 매칭 없으면 nil.
    private static func categoryForPrefix(_ component: String) -> ParaCategory? {
        if component.hasPrefix("10000_") { return .projects }
        if component.hasPrefix("20000_") { return .areas }
        if component.hasPrefix("30000_") { return .resources }
        if component.hasPrefix("40000_") { return .archive }
        return nil
    }

    /// PARA 분류순(archive 맨 끝) → 폴더 먼저 → 이름 오름차순. 항목 식별자·children 보존.
    static func sorted(_ items: [FileTreeItem]) -> [FileTreeItem] {
        items.sorted { a, b in
            let ra = classify(a.url).sortRank
            let rb = classify(b.url).sortRank
            if ra != rb { return ra < rb }
            if a.isDirectory != b.isDirectory { return a.isDirectory && !b.isDirectory }
            return a.name.localizedStandardCompare(b.name) == .orderedAscending
        }
    }
}
```

```swift
    private var paraCategory: ParaCategory { ParaLens.classify(item.url) }

    /// 분류 스타일이 적용된 행 라벨(아이콘+이름+즐겨찾기 별).
    private var rowLabel: some View {
        HStack(spacing: 4) {
            Image(systemName: item.icon)
                .foregroundStyle(paraCategory == .projects ? Color.cmdsAccent : Color.primary)
            Text(item.name)
                .fontWeight(paraCategory == .projects ? .medium : .regular)
                .lineLimit(1)
            if isFavorited {
                Image(systemName: "star.fill")
                    .font(.caption2)
                    .foregroundColor(.yellow)
            }
        }
        .opacity(paraCategory == .archive ? 0.45 : 1)
    }

    var body: some View {
        if item.isDirectory {
            DisclosureGroup(isExpanded: Binding(
                get: { appState.expandedFolders.contains(item.url) },
                set: { _ in appState.toggleFolderExpansion(item.url) }
            )) {
                ForEach(ParaLens.sorted(item.children)) { child in
                    FileTreeItemRow(item: child)
                }
            } label: {
                rowLabel
            }
            .contextMenu {
                FileTreeContextMenu(item: item)
            }
        } else {
            rowLabel
                .onTapGesture {
                    appState.openDocument(at: item.url, inNewTab: true)
                }
                .contextMenu {
                    FileTreeContextMenu(item: item)
                }
        }
    }
```
